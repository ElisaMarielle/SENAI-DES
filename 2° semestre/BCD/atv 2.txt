CREATE DATABASE teste2;

USE teste2;

CREATE TABLE Usuarios(
	id INTEGER AUTO_INCREMENT PRIMARY KEY,
	nome VARCHAR(50),
	email VARCHAR(50)
);

CREATE TABLE Livros(
	id INTEGER AUTO_INCREMENT PRIMARY KEY,
	titulo VARCHAR(50),
	autor VARCHAR(50),
	ano INT
);

CREATE TABLE Emprestimos(
	id INTEGER AUTO_INCREMENT PRIMARY KEY,
	data_retirada DATE,
	data_devolucao DATE,
	usuario_id INT,
	livro_id INT,
	FOREIGN KEY (usuario_id) REFERENCES Usuarios(id),
	FOREIGN KEY (livro_id) REFERENCES Livros(id)
);
